# 高階関数

```Go
package main

import "fmt"

type Hoge struct {
  str string
  id  int
}

func (h *Hoge) GetStr() string {
  return h.str
}

func (h *Hoge) GetId() int {
  return h.id
}

func HogeContain(array []Hoge, cond func(h Hoge) bool) bool {
  for _, elem := range array {
    if cond(elem) {
      return true
    }
  }
  return false
}

func main() {
 hoges := []Hoge{
  {str: "a", id: 1},
  {str: "b", id: 2},
 }
 key := "a"
 if HogeContain(hoges, func(elem Hoge) bool { return elem.GetStr() == key } {
  fmt.Println("Hoge!!")
 } else {
  fmt.Println("No Hoge!!")
 }
}
```
